POST /api/snippet-create
Upload snippet baru ▼
Snippet key akan di-hash SHA-256 di server sebelum disimpan. Simpan key asli kamu — tidak bisa dipulihkan dari DB.
Body Parameters
authorreqstringNama/username penulis snippet
titlereqstringJudul snippet
descriptionreqstringDeskripsi singkat snippet
languageoptstringJavaScript | TypeScript | Python | PHP | Go (default: JavaScript)
tagsoptstringTags dipisah koma: "fetch, async, api"
codereqstringIsi kode snippet
snippetKeyreqstringKey pribadi 3–7 karakter untuk edit/delete nanti
python
import requests

payload = {
    'author':      'devuser',
    'title':       'Hello World',
    'description': 'Contoh snippet',
    'language':    'Python',
    'tags':        'hello, contoh',
    'code':        'print("Hello, World!")',
    'snippetKey':  'mykey1'
}

res = requests.post('https://your-domain.vercel.app/api/snippet-create', json=payload)
print(res.json())
201 Created
{ "ok": true }

Endpoint admin hanya untuk owner. Jangan share credentials ke siapapun. Session disimpan sebagai HttpOnly cookie.
POST/api/admin-login
Login admin → JWT cookie ▼
Body Parameters
usernamereqstringAdmin username (dari env var ADMIN_USERNAME)
passwordreqstringAdmin password (akan di-hash SHA-256 + salt sebelum dibandingkan)
Jika berhasil, server set admin_token sebagai HttpOnly cookie yang expire dalam 8 jam. Cookie ini tidak bisa dibaca JavaScript — hanya dikirim otomatis di setiap request.

PY Python Scraper
Fetch + filter + export JSON ▼
python
import requests, json

BASE_URL = "https://your-domain.vercel.app"

# ── Ambil semua snippet
def get_all():
    res = requests.get(f"{BASE_URL}/api/snippets")
    res.raise_for_status()
    return res.json()

# ── Filter by language
def filter_lang(snippets, lang):
    return [s for s in snippets if s["language"] == lang]

# ── Export ke file JSON
def export_json(data, filename="snippets.json"):
    with open(filename, "w") as f:
        json.dump(data, f, indent=2, ensure_ascii=False)
    print(f"Saved {len(data)} snippets → {filename}")

# ── Upload snippet baru
def upload(author, title, desc, lang, tags, code, key):
    res = requests.post(f"{BASE_URL}/api/snippet-create", json={
        "author": author, "title": title,
        "description": desc, "language": lang,
        "tags": tags, "code": code, "snippetKey": key
    })
    return res.json()

# ── Main
if __name__ == "__main__":
    all_snippets = get_all()
    print(f"Total: {len(all_snippets)} snippets")

    js_only = filter_lang(all_snippets, "JavaScript")
    print(f"JavaScript: {len(js_only)}")

    export_json(all_snippets)
JS Node.js Scraper
Bulk upload dari file lokal ▼
javascript (node)
// node scraper.js  (Node 18+, tidak perlu install package)

const BASE = 'https://your-domain.vercel.app';

// ── Ambil semua snippet
async function fetchAll() {
  const res = await fetch(`${BASE}/api/snippets`);
  if (!res.ok) throw new Error(`HTTP ${res.status}`);
  return res.json();
}

// ── Upload snippet
async function upload(payload) {
  const res = await fetch(`${BASE}/api/snippet-create`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  });
  return res.json();
}

// ── Bulk upload dari array
async function bulkUpload(items) {
  for (const item of items) {
    const result = await upload(item);
    console.log(`✓ ${item.title}`, result);
    // delay 300ms antar request
    await new Promise(r => setTimeout(r, 300));
  }
}

// ── Main
(async () => {
  const snippets = await fetchAll();
  console.log(`Loaded: ${snippets.length} snippets`);
  console.log('Top liked:', snippets.sort((a,b) => b.likes - a.likes)[0]?.title);
})();
PHP PHP Scraper
cURL wrapper untuk PHP ▼
php
<?php

$BASE_URL = "https://your-domain.vercel.app";

// ── Ambil semua snippet
function getSnippets($base) {
    $ch = curl_init("$base/api/snippets");
    curl_setopt_array($ch, [
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_HTTPHEADER     => ['Accept: application/json']
    ]);
    $body = curl_exec($ch);
    curl_close($ch);
    return json_decode($body, true);
}

// ── Upload snippet
function uploadSnippet($base, $data) {
    $ch = curl_init("$base/api/snippet-create");
    curl_setopt_array($ch, [
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_POST           => true,
        CURLOPT_POSTFIELDS     => json_encode($data),
        CURLOPT_HTTPHEADER     => ['Content-Type: application/json']
    ]);
    $body = curl_exec($ch);
    curl_close($ch);
    return json_decode($body, true);
}

$snippets = getSnippets($BASE_URL);
echo "Total: " . count($snippets) . " snippets\n";
echo json_encode($snippets[0], JSON_PRETTY_PRINT);
